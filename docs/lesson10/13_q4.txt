For the customer that spent the most (in total of their lifetime as a customer) total_amt_usd, how many web_events did they have for each channel?

WITH t1 AS (
   SELECT a.id, a.name, SUM(o.total_amt_usd) tot_spent
   FROM orders o
   JOIN accounts a
   ON a.id = o.account_id
   GROUP BY a.id, a.name
   ORDER BY 3 DESC
   LIMIT 1)
SELECT a.name, w.channel, COUNT(*)
FROM accounts a
JOIN web_events w
ON a.id = w.account_id AND a.id =  (SELECT id FROM t1)
GROUP BY 1, 2
ORDER BY 3 DESC;


MINE:

WITH get_acct_w_most_total_sales AS (select a.id as account_id, SUM(o.total_amt_usd) as totals
            from accounts a
            join orders o
            on o.account_id = a.id
            group by 1
            order by 2 desc
            limit 1)

select w.channel, COUNT(w.channel)
    from web_events w
    where account_id =
        (select account_id from get_acct_w_most_total_sales)
    group by 1
    order by 2 desc
